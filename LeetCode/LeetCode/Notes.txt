5. Остаток от деления числа 

	1.Получить последнюю цифру числа
	int num = 1234;
	while (num > 0)
	{
		int digit = num % 10;   // ← получаем последнюю цифру
		Console.WriteLine(digit);
		num = num / 10;         // ← убираем последнюю цифру
	}

	2.Разворачивать число
	int num = 123;
	int reversed = 0;

	while (num > 0)
	{
		int digit = num % 10;
		reversed = reversed * 10 + digit;
		num = num / 10;
	}
